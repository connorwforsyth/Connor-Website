Attempted recreation with tsx

```tsx

import { useEffect, useRef } from 'react';

const ImageMagnify = () => {
  const imageRef = useRef(null);

  useEffect(() => {
    // Assuming imagezoomsl is a global function available after the script is loaded
    if (typeof imagezoomsl === 'function' && imageRef.current) {
      imagezoomsl(imageRef.current, {
        innerzoommagnifier: true,
        classmagnifier: "round-loope",
        magnifiersize: [300, 300],
        disablewheel: false
      });
    } else {
      document.querySelector('.msg').style.display = 'block';
    }
  }, []);

  return (
    <div>
      <div className="msg" style={{ display: 'none' }}>Zoom plugin not available</div>
      <img ref={imageRef} className="imagemagnify" src="your-image-source.jpg" alt="Magnifiable" />
    </div>
  );
};

export default ImageMagnify;
```
